const switchDeviceWithRetry = async (
  deviceId: string,
  kind: 'video' | 'audio',
  maxRetries = 3
) => {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const constraints = {
        [kind]: { deviceId: { exact: deviceId } },
      };
      
      const stream = await navigator.mediaDevices.getUserMedia(constraints);
      return stream;
    } catch (err) {
      console.warn(`Retry ${i + 1}/${maxRetries} failed:`, err);
      
      if (i < maxRetries - 1) {
        await new Promise(resolve => setTimeout(resolve, 1000)); // Wait 1s
      }
    }
  }
  
  throw new Error(`Failed to switch ${kind} device after ${maxRetries} retries`);
};
